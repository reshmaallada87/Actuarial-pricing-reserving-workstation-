# report_generator.py
from reportlab.lib.pagesizes import letter
from reportlab.platypus import SimpleDocTemplate, Paragraph, Spacer, Table, TableStyle
from reportlab.lib.styles import getSampleStyleSheet, ParagraphStyle
from reportlab.lib import colors
import io

def generate_pdf_report(customer_info: dict, risk_info: dict, premium_info: dict, reserve_info: dict, explanation: str) -> bytes:
    """
    Generates a professional motor insurance assessment PDF report using ReportLab.
    Returns the PDF as a byte stream.
    """
    buffer = io.BytesIO()
    doc = SimpleDocTemplate(
        buffer,
        pagesize=letter,
        rightMargin=40,
        leftMargin=40,
        topMargin=40,
        bottomMargin=40
    )
    
    styles = getSampleStyleSheet()
    
    # Custom Styles
    title_style = ParagraphStyle(
        'DocTitle',
        parent=styles['Heading1'],
        fontSize=22,
        leading=26,
        textColor=colors.HexColor('#1E3A8A'), # Deep Blue
        spaceAfter=15
    )
    
    section_heading = ParagraphStyle(
        'SecHeading',
        parent=styles['Heading2'],
        fontSize=14,
        leading=18,
        textColor=colors.HexColor('#0F172A'), # Charcoal
        spaceBefore=12,
        spaceAfter=6,
        keepWithNext=True
    )
    
    body_style = ParagraphStyle(
        'DocBody',
        parent=styles['Normal'],
        fontSize=10,
        leading=14,
        textColor=colors.HexColor('#334155'), # Slate Blue
        spaceAfter=8
    )
    
    bold_body_style = ParagraphStyle(
        'DocBoldBody',
        parent=body_style,
        fontName='Helvetica-Bold'
    )
    
    meta_style = ParagraphStyle(
        'DocMeta',
        parent=styles['Normal'],
        fontSize=8,
        leading=11,
        textColor=colors.HexColor('#64748B'),
        spaceAfter=4
    )

    story = []
    
    # Header Banner / Logo Placeholder
    story.append(Paragraph("MOTOR INSURANCE RISK ASSESSMENT & RESERVING REPORT", title_style))
    story.append(Paragraph("Generated by: AI Insurance Underwriter & Reserving Agent", meta_style))
    story.append(Paragraph("Confidential Underwriting Document • For Internal Professional Use Only", meta_style))
    story.append(Spacer(1, 15))
    
    # 1. Customer & Vehicle Information
    story.append(Paragraph("1. Customer & Vehicle Information", section_heading))
    info_data = [
        [Paragraph("Driver Age", bold_body_style), Paragraph(f"{customer_info['driver_age']} Years", body_style),
         Paragraph("Vehicle Type", bold_body_style), Paragraph(f"{customer_info['vehicle_type']}", body_style)],
        [Paragraph("Vehicle Age", bold_body_style), Paragraph(f"{customer_info['vehicle_age']} Years", body_style),
         Paragraph("Annual Mileage", bold_body_style), Paragraph(f"{customer_info['annual_mileage']:,} KM", body_style)],
        [Paragraph("Previous Claims", bold_body_style), Paragraph(f"{customer_info['previous_claims']}", body_style),
         Paragraph("Claim Type", bold_body_style), Paragraph(f"{customer_info['claim_type']}", body_style)],
        [Paragraph("Est. Repair Cost", bold_body_style), Paragraph(f"₹{customer_info['estimated_repair_cost']:,.2f}", body_style),
         Paragraph("", body_style), Paragraph("", body_style)]
    ]
    
    t_info = Table(info_data, colWidths=[120, 140, 120, 140])
    t_info.setStyle(TableStyle([
        ('BACKGROUND', (0,0), (-1,-1), colors.HexColor('#F8FAFC')),
        ('BOX', (0,0), (-1,-1), 1, colors.HexColor('#E2E8F0')),
        ('INNERGRID', (0,0), (-1,-1), 0.5, colors.HexColor('#F1F5F9')),
        ('PADDING', (0,0), (-1,-1), 6),
        ('VALIGN', (0,0), (-1,-1), 'MIDDLE'),
    ]))
    story.append(t_info)
    story.append(Spacer(1, 12))
    
    # 2. Risk Scoring Analysis
    story.append(Paragraph("2. Actuarial Risk Assessment", section_heading))
    
    risk_color = colors.HexColor('#16A34A') if risk_info['category'] == "Low Risk" else (
        colors.HexColor('#D97706') if risk_info['category'] == "Medium Risk" else colors.HexColor('#DC2626')
    )
    
    risk_data = [
        [Paragraph("Risk Evaluation Index", bold_body_style), Paragraph("Score Breakdown / Weighting", bold_body_style)],
        [
            Paragraph(f"<b>Risk Score:</b> {risk_info['total_score']} / 90<br/><b>Risk Class:</b> <font color='{risk_color}'><b>{risk_info['category']}</b></font>", body_style),
            Paragraph(
                f"• Driver Age Surcharge: {risk_info['breakdown']['driver_age']} pts<br/>"
                f"• Vehicle Age Depreciation Risk: {risk_info['breakdown']['vehicle_age']} pts<br/>"
                f"• Historical Claims Loading: {risk_info['breakdown']['previous_claims']} pts<br/>"
                f"• Exposure (Annual Mileage): {risk_info['breakdown']['annual_mileage']} pts",
                body_style
            )
        ]
    ]
    t_risk = Table(risk_data, colWidths=[200, 320])
    t_risk.setStyle(TableStyle([
        ('BACKGROUND', (0,0), (-1,0), colors.HexColor('#F1F5F9')),
        ('BOX', (0,0), (-1,-1), 1, colors.HexColor('#CBD5E1')),
        ('INNERGRID', (0,0), (-1,-1), 0.5, colors.HexColor('#E2E8F0')),
        ('PADDING', (0,0), (-1,-1), 8),
        ('VALIGN', (0,0), (-1,-1), 'TOP'),
    ]))
    story.append(t_risk)
    story.append(Spacer(1, 12))
    
    # 3. Premium Pricing Recommendation
    story.append(Paragraph("3. Premium Pricing Recommendations", section_heading))
    pricing_data = [
        [Paragraph("Pricing Factor / Adjustment Component", bold_body_style), Paragraph("Surcharge %", bold_body_style), Paragraph("Impact Amount", bold_body_style)],
        [Paragraph("Base Standard Risk Premium", body_style), Paragraph("—", body_style), Paragraph(f"₹{premium_info['base_premium']:,.2f}", body_style)]
    ]
    
    for adj in premium_info['adjustments']:
        pricing_data.append([
            Paragraph(adj['factor'], body_style),
            Paragraph(adj['percentage'], body_style),
            Paragraph(f"+ ₹{adj['amount']:,.2f}", body_style)
        ])
        
    pricing_data.append([
        Paragraph("<b>Total Recommended Premium</b>", bold_body_style),
        Paragraph("<b>—</b>", bold_body_style),
        Paragraph(f"<b>₹{premium_info['final_premium']:,.2f}</b>", bold_body_style)
    ])
    
    t_price = Table(pricing_data, colWidths=[300, 100, 120])
    t_price.setStyle(TableStyle([
        ('BACKGROUND', (0,0), (-1,0), colors.HexColor('#F1F5F9')),
        ('BACKGROUND', (0,-1), (-1,-1), colors.HexColor('#E2E8F0')),
        ('BOX', (0,0), (-1,-1), 1, colors.HexColor('#CBD5E1')),
        ('INNERGRID', (0,0), (-1,-1), 0.5, colors.HexColor('#E2E8F0')),
        ('PADDING', (0,0), (-1,-1), 6),
        ('VALIGN', (0,0), (-1,-1), 'MIDDLE'),
        ('ALIGN', (1,0), (-1,-1), 'CENTER'),
    ]))
    story.append(t_price)
    story.append(Spacer(1, 12))
    
    # 4. Claims Reserving Analysis
    story.append(Paragraph("4. Outstanding Claims Reserving", section_heading))
    reserve_data = [
        [Paragraph("Claims Metric", bold_body_style), Paragraph("Actuarial Reserve Policy Details", bold_body_style)],
        [Paragraph("Reported Claim Type", body_style), Paragraph(f"{reserve_info['claim_type']} Severity", body_style)],
        [Paragraph("Estimated Repair Cost", body_style), Paragraph(f"₹{reserve_info['estimated_repair_cost']:,.2f}", body_style)],
        [Paragraph("Regulatory Minimum Threshold", body_style), Paragraph(f"₹{reserve_info['minimum_reserve_threshold']:,.2f}", body_style)],
        [Paragraph("<b>Estimated Claim Reserve</b>", bold_body_style), Paragraph(f"<b>₹{reserve_info['reserve_amount']:,.2f}</b>", bold_body_style)],
    ]
    t_reserve = Table(reserve_data, colWidths=[200, 320])
    t_reserve.setStyle(TableStyle([
        ('BACKGROUND', (0,0), (-1,0), colors.HexColor('#F1F5F9')),
        ('BACKGROUND', (0,4), (-1,4), colors.HexColor('#FEF08A')), # light yellow highlight for reserve
        ('BOX', (0,0), (-1,-1), 1, colors.HexColor('#CBD5E1')),
        ('INNERGRID', (0,0), (-1,-1), 0.5, colors.HexColor('#E2E8F0')),
        ('PADDING', (0,0), (-1,-1), 6),
        ('VALIGN', (0,0), (-1,-1), 'MIDDLE'),
    ]))
    story.append(t_reserve)
    story.append(Paragraph(f"<b>Rationale:</b> {reserve_info['rationale']}", body_style))
    story.append(Spacer(1, 12))
    
    # 5. Narrative Explanation Agent
    story.append(Paragraph("5. Underwriting Narrative & Decision Explanation", section_heading))
    story.append(Paragraph(explanation, body_style))
    
    # Build Document
    doc.build(story)
    pdf_bytes = buffer.getvalue()
    buffer.close()
    return pdf_bytes
